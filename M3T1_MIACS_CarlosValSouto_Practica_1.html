from torch import nn

class NNModel(nn.Module):
    '''Esta clase define la estructura de la red neuronal:'''
    def __init__(self):
        '''El constructor __init__(self) define qu componentes tiene la red.'''
        
        super(NNModel, self).__init__()
        '''Inicializa la clase padre (nn.Module) que toma del import de la librera "torch" en la primera lnea'''
        
        self.flatten = nn.Flatten() 
        '''Este paso convierte la imagen de 3D a un vector 1D'''
        
        self.fc1 = nn.Linear(224 * 224 * 3, 512)
        '''Crea la primera capa oculta de 512 neuronas. Con "self.fc1", "fc1" se usa por convencin 
        y significa "fully connected". Cada neurona recibe todos los pxeles de la capa de entrada 
        pero cada una analiza un patrn especfico y diferente de las otras.''' 
        
        self.relu1 = nn.ReLU()
        '''Los datos pasan luego a la "Funcin de activacin 'no-lineal' ReLU" la cual decide 
        si una neurona se "activa" o no'''
        
        self.fc2 = nn.Linear(512,256)
        '''Segunda capa oculta, "fc2" crea una segunda capa oculta "fully connected". Esta recibe
        los datos de las 512 neuronas de la primera capa oculta.  '''
        self.relu2 = nn.ReLU()
        '''El mismo paso que el anterior'''
        
        self.fc3 = nn.Linear(256, num_classes)
        '''Crea una capa de 4 neuronas con "num_classes", y es tambin la capa de salida: Toma los datos 
        de la segunda capa oculta de 256 neuronas. No es necesario usar ReLU.  '''
        
        

    def forward(self, x):
        '''Este mtodo define cmo tiene que ser el flujo de informacin a travs de la red'''
        x = self.flatten(x)
        
        x = self.fc1(x)
        x = self.relu1(x)
        
        x = self.fc2(x)
        x = self.relu2(x)
        
        x = self.fc3(x)
        
        return x

def training(dataloader, model, loss_fn, optimizer):

    model.train() # Indicamos al modelo que vamos a entrenar
    
    # variables para acumular mtricas
    perdida_total = 0.0
    predicciones_correctas = 0
    muestras_totales = 0
    
    # Iterar por todos los batches del dataloader
    for indice_batch, (imagenes, etiquetas) in enumerate(dataloader):
        # Mover datos a GPU/CPU segn corresponda
        imagenes, etiquetas = imagenes.to(device), etiquetas.to(device)
        
        # Forward pass: obtener predicciones
        predicciones = model(imagenes)
        
        # Calcular prdida
        loss = loss_fn(predicciones, etiquetas)
        
        # Backward pass: calcular gradientes
        optimizer.zero_grad()  # Limpiar gradientes previos
        loss.backward()        # Calcular gradientes
        optimizer.step()       # Actualizar pesos
        
        # Calcular mtricas
        perdida_total += loss.item()
        _, clases_predichas = torch.max(predicciones, 1)
        predicciones_correctas += (clases_predichas == etiquetas).sum().item()
        muestras_totales += etiquetas.size(0)
    
    # Calcular mtricas promedio
    avg_train_loss = perdida_total / len(dataloader)
    accuracy = 100.0 * predicciones_correctas / muestras_totales
    
    print(f"Train Metrics: \n Accuracy: {accuracy:>0.1f}%, Avg train loss: {avg_train_loss:>8f}\n")
    return avg_train_loss, accuracy

def test_and_validation(dataloader, model, loss_fn):

    model.eval() # Indicamos que no vamos a entrenar

    perdida_total = 0.0
    predicciones_correctas = 0
    muestras_totales = 0
    
    # Sin clculo de gradientes porque solo evaluamos
    with torch.no_grad():
        
        for imagenes, etiquetas in dataloader: # Iterar por todos los batches del dataloader
           
            imagenes, etiquetas = imagenes.to(device), etiquetas.to(device)  # Mover datos a GPU/CPU segn corresponda
            predicciones = model(imagenes) # Forward pass SOLAMENTE (sin backward)
            loss = loss_fn(predicciones, etiquetas) # Calcular prdida (solo para mtricas)
            
            # Acumular mtricas
            perdida_total += loss.item()
            _, clases_predichas = torch.max(predicciones, 1)
            predicciones_correctas += (clases_predichas == etiquetas).sum().item()
            muestras_totales += etiquetas.size(0)
    
    # Calcular mtricas promedio
    avg_loss = perdida_total / len(dataloader)
    accuracy = 100.0 * predicciones_correctas / muestras_totales
    
    print(f"Val or Test Metrics: \n Accuracy: {accuracy:>0.1f}%, Avg loss: {avg_loss:>8f} \n")
    return avg_loss, accuracy

class AlexNet(nn.Module): 
    '''La clase hereda funcionalidad del algoritmo de "nn" redes neuronales de la librera de torch. '''

    def __init__(self):
        super(AlexNet, self).__init__()
        '''Define el constructor de la clase. '''

        ''' PARTE CONVOLUCIONAL: Aqu se van definiendo las capas de la CNN.    '''
        # Capa convolucional 1: 3  96 canales
        '''
        --------------------------------------------------------------------------------------------------------------------
        | - 3 canales de entrada (RGB). 
        | - 96 canales de salida (96 filtros). Tamao del filtro 11x11 pxeles. Cada filtro detecta un patrn diferente,
        |     como bordes verticales, bordes horizontales, etc.
        | - stride: define el tamao del escaneo. El algoritmo escanea cada imagen a modo de 4 pxeles cada vez. entonces,
        |     cada "pieza de escaneo de 11x11" se superpone a la anterior. En teoria esto hace que suavice las transiciones, 
        |     ya que tiene mejor covertura y lo hace ms robusto al detectar patrones que aparecen entre posiciones. 
        | - padding: aade 2 pxeles de borde con valor 'cero'. Esto sirve para:
        |     - para conservar el tamao de la imagen
        |     - todos los pxeles originales reciben igual atencin. 
        --------------------------------------------------------------------------------------------------------------------
        | ReLU es la funcin de activacin.
        | MaxPool2d reduce el tamao tomando el valor mximo de cada regin.
        --------------------------------------------------------------------------------------------------------------------   '''
        
        
        self.conv1 = nn.Conv2d(3, 96, kernel_size=11, stride=4, padding=2)
        self.relu1 = nn.ReLU()
        self.maxpool1 = nn.MaxPool2d(kernel_size=3, stride=2)
        
        
        # Capa convolucional 2: 96  256 canales
        '''La siguiente capa, la segunda, se definen de forma parecida.
        --------------------------------------------------------------------------------------------------------------------
        | - 96 canales de entrada (resultado del bloque anterior).
        | - 256 canales de salida (256 filtros diferentes). Tamao del filtro 5x5 pxeles ms pequeo que el anterior.
        | - stride: no hay, por lo que usa el valor por defecto stride=1.
        | - padding=2: aade 2 pxeles de borde.
        | - MaxPool2d: segunda reduccin de tamao, comprime/reduce la informacin espacial.
        --------------------------------------------------------------------------------------------------------------------         '''
        self.conv2 = nn.Conv2d(96, 256, kernel_size=5, padding=2)
        self.relu2 = nn.ReLU()
        self.maxpool2 = nn.MaxPool2d(kernel_size=3, stride=2)
        
        '''Siguiendo la arquitectura de AlexNet las capas 3 y 4 son capas intermedias sin la implementacin de MaxPool2d, por lo tanto,
        slamente procesan caractersticas sin reducir ms.  
        --------------------------------------------------------------------------------------------------------------------
        | Capa 3: 256  384 canales. Kernel 3x3 (ms pequeo, para detectar detalles finos).
        | - padding=1: mantiene dimensiones espaciales exactas al usar kernel 3x3.
        | - Sin MaxPool: conserva la resolucin espacial para procesamiento detallado.
        |
        | Capa 4: 384  384 canales (mismo nmero de entrada y salida).
        | - Procesa y refina las caractersticas extradas en la capa anterior.
        | - Permite mayor profundidad en la red sin reducir ms la informacin espacial.
        --------------------------------------------------------------------------------------------------------------------   '''
        # Capa convolucional 3: 256  384 canales
        self.conv3 = nn.Conv2d(256, 384, kernel_size=3, padding=1)
        self.relu3 = nn.ReLU()
        
        # Capa convolucional 4: 384  384 canales
        self.conv4 = nn.Conv2d(384, 384, kernel_size=3, padding=1)
        self.relu4 = nn.ReLU()
        
        # Capa convolucional 5: 384  256 canales
        '''ltima capa. Se aplica por ltima vez la reduccin con MaxPool2d. 
        --------------------------------------------------------------------------------------------------------------------
        | - 384  256 canales: reduce el nmero de canales antes de pasar a las capas densas.
        | - MaxPool2d final: ltima reduccin espacial que prepara los datos para el aplanamiento.
        | - Esta reduccin final da como resultado mapas de caractersticas de 6x6 pxeles por canal.
        --------------------------------------------------------------------------------------------------------------------   '''
        self.conv5 = nn.Conv2d(384, 256, kernel_size=3, padding=1)
        self.relu5 = nn.ReLU()
        self.maxpool3 = nn.MaxPool2d(kernel_size=3, stride=2)
        
        '''A partir de aqu pasamos los datos a una estructura del estilo de red neuronal clsica. 
        Empezamos por el aplanamiento de los datos.  
        --------------------------------------------------------------------------------------------------------------------
        | - Flatten: convierte los mapas 2D (256 mapas de 6x6) en un vector 1D de 256*6*6 = 9,216 elementos.
        | - Esta transformacin permite que las capas fully-connected procesen la informacin extrada por las 
        |   capas convolucionales.
        --------------------------------------------------------------------------------------------------------------------   '''
        # Aplanamiento de los datos que obtenemos de la CNN para alimentar la NN
        self.flatten = nn.Flatten()
        
        '''Se incorpora el DropOut, sigue siendo una red FC (capa densa). La red es tecnicamente "fully connected" pero se implementa
        el "Dropout", el cual apaga aleatoriamente el 50% de neuronas durante el entrenamiento que le indicamos explcitamente 
        con "nn.Dropout(0.5)". Es una tcnica para prevenir el Overfitting.
        --------------------------------------------------------------------------------------------------------------------
        | Capa 1:
        | - Entrada: 9,216 caractersticas (256*6*6 del aplanamiento)
        | - Salida: 4,096 neuronas (reduccin significativa)
        | - Dropout(0.5): durante entrenamiento, aleatoriamente "apaga" 50% de las neuronas para evitar dependencias excesivas 
        |    y mejorar la generalizacin.
        --------------------------------------------------------------------------------------------------------------------   '''
        # Capa densa 1: 25666  4096 neuronas
        self.fc1 = nn.Linear(256 * 6 * 6, 4096)
        self.relu6 = nn.ReLU()
        self.dropout1 = nn.Dropout(0.5)
        
        '''Repite el paso anterior
        --------------------------------------------------------------------------------------------------------------------
        | Capa 2:
        | - Entrada: 4,096 neuronas
        | - Salida: 4,096 neuronas (mismo tamao, refinamiento de caractersticas)
        | - Segundo Dropout: doble proteccin contra overfitting
        --------------------------------------------------------------------------------------------------------------------   '''
        # Capa densa 2: 4096  4096 neuronas
        self.fc2 = nn.Linear(4096, 4096)
        self.relu7 = nn.ReLU()
        self.dropout2 = nn.Dropout(0.5)
                
        '''La capa de salida
        --------------------------------------------------------------------------------------------------------------------
        | - Entrada: 4,096 caractersticas procesadas
        | - Salida: num_classes neuronas (4 en nuestro caso: Angry, Other, Sad, happy)
        | - Sin ReLU: las puntuaciones "crudas" (logits) se pasan directamente a CrossEntropyLoss
        | - Cada neurona de salida representa la "confianza" del modelo en cada clase emocional
        --------------------------------------------------------------------------------------------------------------------   '''
        # Capa de salida: 4096  num_classes
        self.fc3 = nn.Linear(4096, num_classes)

    def forward(self, x):
        ''' Igual que el ejercicio anterior, aqu se define el flujo de informacin en este caso a travs de la red AlexNet.
        Conecta las dos arquitecturas: CNN  Flatten  FC    '''
        
        # Arquitectura CNN: Procesamiento convolucional
        x = self.maxpool1(self.relu1(self.conv1(x)))      # Bloque 1: (batch, 3, 224, 224)  (batch, 96, 27, 27)
        x = self.maxpool2(self.relu2(self.conv2(x)))      # Bloque 2: (batch, 96, 27, 27)  (batch, 256, 13, 13)
        x = self.relu3(self.conv3(x))                     # Bloque 3: (batch, 256, 13, 13)  (batch, 384, 13, 13)
        x = self.relu4(self.conv4(x))                     # Bloque 4: (batch, 384, 13, 13)  (batch, 384, 13, 13)
        x = self.maxpool3(self.relu5(self.conv5(x)))      # Bloque 5: (batch, 384, 13, 13)  (batch, 256, 6, 6)
        
        # Transicin: De mapas 2D a vector 1D
        x = self.flatten(x)                               # (batch, 256, 6, 6)  (batch, 9216)
        
        # Arquitectura FC: Clasificacin
        x = self.dropout1(self.relu6(self.fc1(x)))        # FC1: (batch, 9216)  (batch, 4096)
        x = self.dropout2(self.relu7(self.fc2(x)))        # FC2: (batch, 4096)  (batch, 4096)
        x = self.fc3(x)                                   # Salida: (batch, 4096)  (batch, 4)
        
        return x
